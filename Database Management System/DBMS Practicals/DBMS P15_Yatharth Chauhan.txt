DECLARE
   CURSOR C1 IS

   SELECT EMP_NO,EMP_NAME,L_NAME,EMP_SAL,DEPT_NAME FROM EMPLOYEE WHERE EMP_NO=101;

   EMP_RECORD C1%ROWTYPE;
 
BEGIN

   OPEN C1;
 
LOOP

FETCH C1 INTO EMP_RECORD;
EXIT WHEN C1%NOTFOUND;

DBMS_OUTPUT.PUT_LINE('EMPLOYEE NO IS: ' || EMP_RECORD.EMP_NO);
DBMS_OUTPUT.PUT_LINE(EMP_RECORD.EMP_NAME);
DBMS_OUTPUT.PUT_LINE(EMP_RECORD.L_NAME);
DBMS_OUTPUT.PUT_LINE(EMP_RECORD.EMP_SAL);
DBMS_OUTPUT.PUT_LINE(EMP_RECORD.DEPT_NAME);

END LOOP;

CLOSE C1;

END;
